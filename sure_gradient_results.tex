\section{Scalar case ($n=1$)}

\noindent\textbf{SURE formula  (n=1)}
\begin{align}
  {\left(x-D\left(x\right)\right)}^2+2\,\sigma ^2\,\frac{\partial }{\partial x} D\left(x\right)-\sigma ^2 \tag{\texttt{sure-scalar}}
\end{align}

\noindent\textbf{Full $\nabla_x \mathrm{SURE}$  (n=1, gradient())}
\begin{align}
  2\,\sigma ^2\,\frac{\partial ^2}{\partial x^2} D\left(x\right)-2\,\left(x-D\left(x\right)\right)\,\left(\frac{\partial }{\partial x} D\left(x\right)-1\right) \tag{\texttt{grad-sure-scalar-full}}
\end{align}

\noindent\textbf{Residual part $\nabla\|r\|^2$  (n=1)}
\begin{align}
  -2\,\left(x-D\left(x\right)\right)\,\left(\frac{\partial }{\partial x} D\left(x\right)-1\right) \tag{\texttt{grad-res-scalar}}
\end{align}

\noindent\textbf{Trace part $\nabla(2\sigma^2 \operatorname{tr}\{J_D\})$  (n=1)}
\begin{align}
  2\,\sigma ^2\,\frac{\partial ^2}{\partial x^2} D\left(x\right) \tag{\texttt{grad-tr-scalar}}
\end{align}

\noindent\textbf{[approx] mode gradient  (n=1): $2r$}
\begin{align}
  2\,x-2\,D\left(x\right) \tag{\texttt{grad-approx-scalar}}
\end{align}

\noindent\textbf{[vjp] mode gradient  (n=1): $2(1-J)r$}
\begin{align}
  -2\,\left(x-D\left(x\right)\right)\,\left(\frac{\partial }{\partial x} D\left(x\right)-1\right) \tag{\texttt{grad-vjp-scalar}}
\end{align}

\noindent\textbf{[full]$-$[vjp]  (n=1):  $2\sigma^2 D''(x)$}
\begin{align}
  2\,\sigma ^2\,\frac{\partial ^2}{\partial x^2} D\left(x\right) \tag{\texttt{diff-full-vjp-scalar}}
\end{align}

\noindent\textbf{[vjp]$-$[approx]  (n=1):  $-2J \cdot r$}
\begin{align}
  -2\,\frac{\partial }{\partial x} D\left(x\right)\,\left(x-D\left(x\right)\right) \tag{\texttt{diff-vjp-approx-scalar}}
\end{align}
